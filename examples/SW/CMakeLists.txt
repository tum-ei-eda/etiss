project(SW_example)


install(DIRECTORY ${CMAKE_CURRENT_LIST_DIR}/
		DESTINATION examples/SW
		FILES_MATCHING
    	PATTERN "*.h"
    	PATTERN "*.cpp"
		PATTERN "*.c"
		PATTERN "*.S"
		PATTERN "*.s"
		PATTERN "README*"
    	PATTERN "Makefile"
        PATTERN "*.cmake"
		PATTERN "*.ld"
    	PATTERN ".svn" EXCLUDE
		PATTERN "*.in" EXCLUDE
		PATTERN "CMakeLists.txt" EXCLUDE
)

# Setup OR1K Makefile
configure_file(
  or1k/Makefile.in
  "${PROJECT_BINARY_DIR}/tobeinstalled/or1k/Makefile"
  @ONLY
 )

install(FILES "${PROJECT_BINARY_DIR}/tobeinstalled/or1k/Makefile"
		DESTINATION examples/SW/or1k
)

install(FILES riscv/CMakeLists.txt
    DESTINATION examples/SW/riscv
)
install(FILES riscv_cpp/CMakeLists.txt
    DESTINATION examples/SW/riscv_cpp
)
